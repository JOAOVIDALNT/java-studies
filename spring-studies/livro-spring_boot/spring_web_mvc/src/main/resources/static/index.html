function propertiesService(filter) { $.ajax({ type : "GET", url : "/api/find/"+filter, data : '$format=json', dataType : 'json', success : function(data) { var total=0; $.each(data, function(d, results) { $("#propTable tbody").append( "" + "" + results.category + "" + "" + results.subcategory + "" + "" + results.name + "" + "" + results.value + "" + "" + results.description + "" + "") total++; }) $("#results").text(total+" found"); } }); };
